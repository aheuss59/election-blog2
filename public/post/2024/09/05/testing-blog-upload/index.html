nat_wide <- 
  nat_data |> 
  select(year, party, pv2p) |>
  pivot_wider(names_from = party, values_from = pv2p) |>
  mutate(winner = ifelse(republican > democrat, "republican", "democrat"))

nat_wide |> 
  group_by(winner) |>
  summarize(races_won = n())
